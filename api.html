Api-email :

 
// url = https://jftv.pythonanywhere.com/email=put@email.com

 
import requests

 
email = input('Enter email : ')

 
sent = requests.get('https://jftv.pythonanywhere.com/email='+email).text

 
if 'True' in sent: print('Email is Not available')

 
else: print('Email is available')
